# Importación de corpus a formato MALLET
# Requiere tener MALLET instalado localmente y ajustar la ruta de $MALLET_HOME.
# Ejecutar este script desde la carpeta:
# 01_hd_clasicas/topic_modeling/mallet

$MALLET_HOME = "C:\ruta\a\mallet"

$corpus = @("femina", "filipinas", "heraldo")

foreach ($c in $corpus) {
  & "$MALLET_HOME\bin\mallet.bat" import-file `
    --input ".\corpus_por_frases\$($c)_por_frases.txt" `
    --output ".\topicos\topics_$($c).mallet" `
    --keep-sequence `
    --remove-stopwords `
    --stoplist-file ".\stop-words.txt" `
    --token-regex "\p{L}+"
}
