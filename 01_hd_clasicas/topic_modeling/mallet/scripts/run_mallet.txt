# Entrenamiento de modelos de tópicos con MALLET
# Requiere tener MALLET instalado localmente y ajustar la ruta de $MALLET_HOME. Se recomienda usar `--random-seed 12345` si se desea que los resultados sean más fácilmente reproducibles entre ejecuciones.
# Ejecutar este script desde la carpeta:
# 01_hd_clasicas/topic_modeling/mallet

$MALLET_HOME = "C:\ruta\a\mallet"

$corpus = @("femina", "filipinas", "heraldo")

foreach ($c in $corpus) {
  & "$MALLET_HOME\bin\mallet.bat" train-topics `
    --input ".\topicos\topics_$($c).mallet" `
    --num-topics 10 `
    --optimize-interval 10 `
    --random-seed 12345 `
    --output-topic-keys ".\claves_topicos\claves_topicos_$($c).txt" `
    --output-doc-topics ".\composicion_documentos\composicion_documentos_$($c).txt"
}
